$services = @("auth-service", "user-service", "career-service")
foreach ($s in $services) {
    Push-Location $s
    Write-Host "==> Building $s" -ForegroundColor Cyan
    mvn -q clean package -DskipTests
    if ($LASTEXITCODE -ne 0) {
        Write-Error "Build FAILED in $s"
        Pop-Location
        break
    }
    Pop-Location
}
